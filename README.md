# djangoProjectCommands
pip install django  # Установка Django
django-admin startproject myproject  # Создание нового проекта Django
cd myproject  # Переход в каталог проекта
python manage.py runserver  # Запуск встроенного сервера разработки

python manage.py startapp myapp  # Создание нового приложения

python manage.py makemigrations 
python manage.py migrate

python manage.py createsuperuser # Создание суперпользователя
python manage.py runserver
http://127.0.0.1:8000/admin/
